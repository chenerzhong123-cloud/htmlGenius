# v0.4 插件多人协同批注 — 实现计划

> **For agentic workers:** REQUIRED SUB-SKILL: 用 superpowers:subagent-driven-development(推荐)或 superpowers:executing-plans 逐 task 执行。步骤用 `- [ ]` 复选框跟踪。

**Goal:** 让 v0.3 Chrome 插件支持多人实时协同批注（SSE 推送 + 团队 token 鉴权 + 多级回复 + 作者删除 + 一键复制全量评论），数据全在自有 FastAPI+SQLite 后端。

**Architecture:** 后端加 SSE 房间推送 + 团队鉴权 + team/parent 字段；插件加 `remote-store`/`sync` 模块与 `storage` mode 分派，content-script 持 EventSource、应用 delta。写走 REST，推走 SSE 单向流。

**Tech Stack:** Python 3.9+ / FastAPI / pydantic v2 / SQLite(WAL) / pytest / httpx / Playwright；插件 Vanilla JS + MV3 + EventSource。

## Global Constraints

- Python `>=3.9`；`from __future__ import annotations` 已在 server 模块。
- 数据自托管：禁用任何第三方协同后端；批注只存自有 SQLite。
- `team_id` 一律由服务端从 token 注入，**绝不取自请求体**。
- 稳定 URL：托管 HTML 的 URL 不含版本号（跨版本重定位依赖）。
- 插件 MV3：长连接(EventSource)住 content-script，不住 background service worker。
- commit message 中文 + conventional 前缀（`feat(v0.4-TN):` / `fix(v0.4-TN):` / `test(v0.4-TN):` / `docs(v0.4):`）。
- 后端测试用 `fastapi.testclient.TestClient` + `storage.init_db(tmp_path/...)` 模式（见 `tests/test_versions_api.py`）；SSE/异步用 `asyncio.run` + `httpx.ASGITransport`（httpx 已在 dev deps）。
- 「仅作者删除」在团队 token 下为软约束（声明 user.id 未加密验证，§5.10），需 OAuth 才硬化。

# server/auth.py
import json
import os
from fastapi import Header, HTTPException, Query


def _teams() -> dict:
    return json.loads(os.environ.get("HG_TEAMS", "{}"))


def require_team(authorization: str | None = Header(None)) -> str:
    token = (authorization or "").removeprefix("Bearer ").strip()
    teams = _teams()
    if token not in teams:
        raise HTTPException(status_code=401, detail="invalid token")
    return teams[token]


def require_team_query(token: str = Query(...)) -> str:
    teams = _teams()
    if token not in teams:
        raise HTTPException(status_code=401, detail="invalid token")
    return teams[token]
```

# server/storage.py —— 替换 delete_annotation:
def delete_annotation(aid: str, team_id: str, actor_id: str) -> list[dict]:
    """作者校验 + 级联删子树。非作者/非本团队抛 PermissionError。返回 [{id, document_id}]。"""
    c = _connect()
    try:
        row = c.execute("SELECT team_id, author, document_id FROM annotations WHERE id=?", (aid,)).fetchone()
        if row is None:
            return []
        if row["team_id"] != team_id:
            raise PermissionError("wrong team")
        if json.loads(row["author"]).get("id") != actor_id:
            raise PermissionError("not owner")
        doc_id = row["document_id"]
        # 收集子树(BFS by parent_id)
        to_delete = [aid]
        queue = [aid]
        while queue:
            cur = queue.pop()
            children = [r["id"] for r in c.execute(
                "SELECT id FROM annotations WHERE parent_id=?", (cur,)
            ).fetchall()]
            to_delete.extend(children)
            queue.extend(children)
        c.execute("BEGIN IMMEDIATE")
        try:
            for did in to_delete:
                c.execute("DELETE FROM annotations WHERE id=?", (did,))
            c.execute("COMMIT")
        except Exception:
            c.execute("ROLLBACK")
            raise
        return [{"id": did, "document_id": doc_id} for did in to_delete]
    finally:
        c.close()
```

# server/sse.py
import asyncio
from collections import defaultdict


class RoomManager:
    def __init__(self) -> None:
        self._queues: dict[tuple[str, str], set[asyncio.Queue]] = defaultdict(set)

    async def subscribe(self, team_id: str, doc_id: str) -> asyncio.Queue:
        q: asyncio.Queue = asyncio.Queue()
        self._queues[(team_id, doc_id)].add(q)
        return q

    def unsubscribe(self, team_id: str, doc_id: str, q: asyncio.Queue) -> None:
        self._queues[(team_id, doc_id)].discard(q)

    async def broadcast(self, team_id: str, doc_id: str, event: str, data: dict) -> None:
        for q in list(self._queues.get((team_id, doc_id), ())):
            await q.put({"event": event, "data": data})


rooms = RoomManager()

                    yield ": ping\n\n"  # keepalive 注释

# server/presence.py
import time
from collections import defaultdict
from .sse import rooms

_TTL = 60  # 秒

# (team_id, doc_id) -> { user_id: {"user": {...}, "last_seen": ts} }
_USERS: dict[tuple[str, str], dict[str, dict]] = defaultdict(dict)


def _gc(key: tuple[str, str]) -> None:
    now = time.time()
    for uid in [u for u, v in _USERS[key].items() if now - v["last_seen"] > _TTL]:
        _USERS[key].pop(uid, None)


async def update(team_id: str, doc: str, user: dict, op: str) -> None:
    key = (team_id, doc)
    _gc(key)
    uid = user.get("id")
    if op == "bye":
        _USERS[key].pop(uid, None)
    else:  # join | heartbeat
        _USERS[key][uid] = {"user": user, "last_seen": time.time()}
    users = [v["user"] for v in _USERS[key].values()]
    await rooms.broadcast(team_id, doc, "presence", {"users": users})

window.RemoteStore = (function () {
  function authHeaders(cfg) {
    return {
      "Authorization": "Bearer " + cfg.team_token,
      "X-User-Id": (cfg.user && cfg.user.id) || "u_self",
      "X-User-Name": (cfg.user && cfg.user.name) || "作者",
      "Content-Type": "application/json",
    };
  }

  function buildCreateRequest(cfg, docId, ann) {
    return {
      url: cfg.backend + "/api/annotations",
      headers: authHeaders(cfg),
      body: {
        document_id: docId,
        selector: ann.selector,
        quote: ann.quote,
        body: ann.body || { comment: "", action: "rewrite", instruction: "" },
        parent_id: ann.parent_id || null,
      },
    };
  }
  function parseCreate(json) { return json; }

  function make(cfg) {
    return {
      async getDocumentId() { return location.origin + location.pathname; },
      async saveAnnotation(ann) {
        const docId = ann.document_id || await this.getDocumentId();
        const req = buildCreateRequest(cfg, docId, ann);
        const r = await fetch(req.url, { method: "POST", headers: req.headers, body: JSON.stringify(req.body) });
        if (!r.ok) throw new Error("save failed " + r.status);
        return r.json();
      },
      async listAnnotations(docId) {
        const r = await fetch(cfg.backend + "/api/annotations?document_id=" + encodeURIComponent(docId),
          { headers: authHeaders(cfg) });
        if (!r.ok) throw new Error("list failed " + r.status);
        const j = await r.json();

      },
      async deleteAnnotation(id) {
        const r = await fetch(cfg.backend + "/api/annotations/" + encodeURIComponent(id),
          { method: "DELETE", headers: authHeaders(cfg) });
        if (!r.ok && r.status !== 403) throw new Error("delete failed " + r.status);
        return r.ok;
      },
    };
  }

// extension/sync.js — EventSource 客户端 + delta 解析 + presence
window.Sync = (function () {
  // 纯函数:SSE event/data → delta(测试用)
  function parseEvent(event, dataStr) {
    let data = {};
    try { data = JSON.parse(dataStr || "{}"); } catch (e) { data = {}; }
    if (event === "annotation:created") return { op: "create", annotation: data };
    if (event === "annotation:deleted") return { op: "delete", id: data.id };
    if (event === "presence") return { op: "presence", users: data.users || [] };
    return null;  // hello 等忽略
  }

  function start(opts) {
    const url = opts.backend + "/api/stream?doc=" + encodeURIComponent(opts.docId)
              + "&token=" + encodeURIComponent(opts.team_token);
    const es = new EventSource(url);
    let hbTimer = null;

    function sendPresence(op) {
      fetch(opts.backend + "/api/presence", {
        method: "POST",
        headers: { "Authorization": "Bearer " + opts.team_token, "Content-Type": "application/json",
                   "X-User-Id": opts.user.id, "X-User-Name": opts.user.name || "" },
        body: JSON.stringify({ doc: opts.docId, user: opts.user, op: op || "heartbeat" }),
      }).catch(() => {});
    }

    es.addEventListener("annotation:created", (e) => { opts.onCreate && opts.onCreate(JSON.parse(e.data)); });
    es.addEventListener("annotation:deleted", (e) => { opts.onDelete && opts.onDelete(JSON.parse(e.data).id); });
    es.addEventListener("presence", (e) => { opts.onPresence && opts.onPresence(JSON.parse(e.data).users); });
    es.onopen = () => { sendPresence("join"); hbTimer = setInterval(() => sendPresence("heartbeat"), 25000); };
    es.onerror = () => { /* EventSource 自动重连 */ };

    function stop() {
      if (hbTimer) clearInterval(hbTimer);
      try { navigator.sendBeacon(opts.backend + "/api/presence",
        new Blob([JSON.stringify({ doc: opts.docId, user: opts.user, op: "bye" })], { type: "application/json" })); } catch (e) {}
      es.close();
    }

    return { stop, sendPresence };
  }

  return { start, parseEvent };

```javascript
// extension/buildprompt.js — 多级回复 DFS 组装(从 sidepanel.js exportPrompt 抽出并增强)
window.BuildPrompt = (function () {
  function buildTree(items) {
    const byParent = {};
    items.forEach((a) => {
      const p = a.parent_id || null;
      (byParent[p] = byParent[p] || []).push(a);
    });
    return byParent;
  }

  function fmtBlock(ann, depth) {
    const sel = ann.selector || {};
    const exact = sel.exact || ann.quote || "";
    const prefix = (sel.prefix || "").trim();
    const suffix = (sel.suffix || "").trim();
    const indent = "  ".repeat(depth);
    const loc = (prefix || suffix)
      ? "定位:" + (prefix ? "前文「" + prefix + "」 " : "") + "【原文】「" + exact + "」" + (suffix ? " 后文「" + suffix + "」" : "")
      : (exact ? "定位:【原文】「" + exact + "」" : "定位:(回复,无独立选区)");
    const who = (ann.author && ann.author.name) ? "[" + ann.author.name + "] " : "";
    const c = (ann.body && ann.body.comment) || "(无)";
    return indent + loc + "\n" + indent + "评论:" + who + c;
  }

  function dfs(node, byParent, depth, out) {
    (byParent[node.id] || []).forEach((child) => {
      out.push(fmtBlock(child, depth));
      dfs(child, byParent, depth + 1, out);
    });
  }

  function fromAnnotations(items) {
    const byParent = buildTree(items || []);
    const out = [];
    let n = 0;
    (byParent[null] || []).forEach((top) => {
      n += 1;
      out.push("==批注" + n + "==");
      out.push(fmtBlock(top, 0));
      dfs(top, byParent, 1, out);
    });
    const header = "你是一名 HTML 编辑执行器。下面给出文档的全部 " + n + " 条顶层批注(含多级回复),请逐条执行修改,并输出完整的新版 HTML:\n\n";
    return header + out.join("\n");
  }

  return { fromAnnotations, buildTree, fmtBlock };
})();
```
